		$(function(){
			$buscaSubMenu = $('#BUSCA');
			$buscaSubMenu.focus(function(){
				if($(this).val() == "Digite uma localidade"){
					$(this).val("");
				}
			}).blur(function(){
				if($(this).val() == ""){
					$(this).val("Digite uma localidade");
				}
			});

			$('form[name="buscaMenu"]').submit(function(event){
				if($buscaSubMenu.val().length > 2 && $buscaSubMenu.val().indexOf('Digite uma localidade') < 0){
					window.location = '/busca/' + escape($buscaSubMenu.val());
				}
				return false;
				event.preventDefault();
			});
		});
	</script>
